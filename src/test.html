const LOG_CONFIG = {
	debug: true, 
	level: Log4JS.LEVELS.TRACE, 
	file: "log4js.log",
	prefix: "MyLogger>"
	date_format: "HH:MM:ss"
};
var log = new Log4JS(LOG_CONFIG);
log.debug(LOG4JS_HEADER);
log.info("Info message");
log.warn(["Grouped messages", "First message", "Second message"]);
